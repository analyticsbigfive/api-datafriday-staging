// Debug endpoint to check all keys
app.get("/make-server-eb31619c/debug/keys", async (c) => {
  try {
    const spaces = await kv.getByPrefix("space:");
    const configs = await kv.getByPrefix("config:");
    
    // Group configurations by spaceId
    const configsBySpace: Record<string, any[]> = {};
    const orphanedConfigs: any[] = [];
    
    configs.forEach((config: any) => {
      if (config.spaceId) {
        if (!configsBySpace[config.spaceId]) {
          configsBySpace[config.spaceId] = [];
        }
        configsBySpace[config.spaceId].push(config);
      } else {
        orphanedConfigs.push(config);
      }
    });
    
    return c.json({ 
      success: true, 
      data: {
        spacesCount: spaces.length,
        spaces: spaces.map((s: any) => ({ 
          id: s.id, 
          name: s.name,
          configCount: configsBySpace[s.id]?.length || 0
        })),
        configurationsCount: configs.length,
        configurations: configs.map((c: any) => ({ 
          id: c.id, 
          name: c.name, 
          spaceId: c.spaceId,
          hasData: !!(c.data)
        })),
        configsBySpace,
        orphanedConfigs: orphanedConfigs.map((c: any) => ({
          id: c.id,
          name: c.name
        }))
      }
    });
  } catch (error) {
    console.error("Error fetching debug keys:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Deduplicate market prices
app.post("/make-server-eb31619c/market-prices/deduplicate", async (c) => {
  try {
    const prices = await kv.getByPrefix("market-price:");
    const seen = new Map<string, any>();
    let deletedCount = 0;
    
    for (const price of prices) {
      // Create a key based on all fields except id
      const key = JSON.stringify({
        supplier_id: price.supplier_id || price.supplier, // Support both old and new format
        itemName: price.itemName,
        goodType: price.goodType,
        unit: price.unit,
        unitsPerPurchase: price.unitsPerPurchase,
        price: price.price,
        pricePerUnit: price.pricePerUnit,
      });
      
      if (seen.has(key)) {
        // Duplicate found - delete this one
        await kv.del(`market-price:${price.id}`);
        deletedCount++;
        console.log(`Deleted duplicate market price: ${price.itemName} (id: ${price.id})`);
      } else {
        // First occurrence - keep it
        seen.set(key, price);
      }
    }
    
    return c.json({ 
      success: true, 
      message: `Removed ${deletedCount} duplicate market price entries`,
      deletedCount 
    });
  } catch (error) {
    console.error("Error deduplicating market prices:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Create/Update menu component
app.post("/make-server-eb31619c/menu-components", async (c) => {
  try {
    const body = await c.req.json();
    const id = body.id || `menu-component-${Date.now()}`;
    
    // Validate and repair subComponents data
    const repairedSubComponents = body.subComponents?.map((sub: any) => {
      const repairedSub = { ...sub };
      
      // Ensure all required fields are present
      if (!repairedSub.unitCost || repairedSub.unitCost === 0) {
        // If unitCost is missing but we have cost and numberOfUnits, calculate it
        if (repairedSub.cost > 0 && repairedSub.numberOfUnits > 0) {
          repairedSub.unitCost = repairedSub.cost / repairedSub.numberOfUnits;
          console.log(`Repaired unitCost for "${repairedSub.name}": ${repairedSub.unitCost}`);
        }
      }
      
      // Ensure numberOfUnits is present
      if (!repairedSub.numberOfUnits || repairedSub.numberOfUnits === 0) {
        // If numberOfUnits is missing but we have cost and unitCost, calculate it
        if (repairedSub.cost > 0 && repairedSub.unitCost > 0) {
          repairedSub.numberOfUnits = repairedSub.cost / repairedSub.unitCost;
          console.log(`Repaired numberOfUnits for "${repairedSub.name}": ${repairedSub.numberOfUnits}`);
        } else {
          // Default to 1.0 if we can't calculate
          repairedSub.numberOfUnits = 1.0;
          console.log(`Set default numberOfUnits for "${repairedSub.name}": 1.0`);
        }
      }
      
      // Ensure cost is consistent
      repairedSub.cost = (repairedSub.unitCost || 0) * (repairedSub.numberOfUnits || 0);
      
      return repairedSub;
    }) || [];
    
    const data = {
      ...body,
      id,
      subComponents: repairedSubComponents,
    };
    
    console.log(`Saving menu component "${data.name}" (${id}) with ${repairedSubComponents.length} subComponents`);
    await kv.set(`menu-component:${id}`, data);
    return c.json({ success: true, data });
  } catch (error) {
    console.error("Error saving menu component:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Repair all menu components - fix missing numberOfUnits
app.post("/make-server-eb31619c/menu-components/repair", async (c) => {
  try {
    const components = await kv.getByPrefix("menu-component:");
    let repairedCount = 0;
    let totalSubComponentsRepaired = 0;
    
    for (const component of components) {
      let componentNeedsRepair = false;
      
      const repairedSubComponents = component.subComponents?.map((sub: any) => {
        const repairedSub = { ...sub };
        let subNeedsRepair = false;
        
        // Fix unitCost if missing
        if (!repairedSub.unitCost || repairedSub.unitCost === 0) {
          if (repairedSub.cost > 0 && repairedSub.numberOfUnits > 0) {
            repairedSub.unitCost = repairedSub.cost / repairedSub.numberOfUnits;
            subNeedsRepair = true;
            console.log(`Repaired unitCost for "${repairedSub.name}" in component "${component.name}": ${repairedSub.unitCost}`);
          }
        }
        
        // Fix numberOfUnits if missing or zero
        if (!repairedSub.numberOfUnits || repairedSub.numberOfUnits === 0) {
          if (repairedSub.cost > 0 && repairedSub.unitCost > 0) {
            repairedSub.numberOfUnits = repairedSub.cost / repairedSub.unitCost;
            subNeedsRepair = true;
            console.log(`Repaired numberOfUnits for "${repairedSub.name}" in component "${component.name}": ${repairedSub.numberOfUnits}`);
          } else {
            repairedSub.numberOfUnits = 1.0;
            subNeedsRepair = true;
            console.log(`Set default numberOfUnits for "${repairedSub.name}" in component "${component.name}": 1.0`);
          }
        }
        
        // Recalculate cost to ensure consistency
        repairedSub.cost = (repairedSub.unitCost || 0) * (repairedSub.numberOfUnits || 0);
        
        if (subNeedsRepair) {
          totalSubComponentsRepaired++;
          componentNeedsRepair = true;
        }
        
        return repairedSub;
      }) || [];
      
      if (componentNeedsRepair) {
        const repairedComponent = {
          ...component,
          subComponents: repairedSubComponents,
        };
        await kv.set(`menu-component:${component.id}`, repairedComponent);
        repairedCount++;
        console.log(`Repaired component "${component.name}" (${component.id})`);
      }
    }
    
    return c.json({ 
      success: true, 
      message: `Repaired ${repairedCount} components (${totalSubComponentsRepaired} subComponents total)`,
      repairedCount,
      totalSubComponentsRepaired
    });
  } catch (error) {
    console.error("Error repairing menu components:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Refresh menu item costs based on current ingredient/component prices
app.post("/make-server-eb31619c/menu-items/refresh-costs", async (c) => {
  try {
    const menuItems = await kv.getByPrefix("menu-item:");
    const ingredients = await kv.getByPrefix("ingredient:");
    const components = await kv.getByPrefix("menu-component:");
    
    let updatedCount = 0;
    let totalCostChanges = 0;
    
    for (const menuItem of menuItems) {
      let hasChanges = false;
      
      // Update each component/ingredient cost based on current prices
      const updatedComponents = menuItem.components?.map((comp: any) => {
        if (comp.itemType === 'Ingredient') {
          // Find the current ingredient
          const currentIngredient = ingredients.find((ing: any) => ing.id === comp.sourceId);
          if (currentIngredient) {
            const newUnitCost = currentIngredient.costPerRecipeUnit;
            const newStorageType = currentIngredient.storageType;
            if (newUnitCost !== comp.unitCost || newStorageType !== comp.storageType) {
              hasChanges = true;
              if (newUnitCost !== comp.unitCost) {
                totalCostChanges++;
                console.log(`Updated cost for ingredient "${comp.name}" in menu item "${menuItem.name}": ${comp.unitCost} → ${newUnitCost}`);
              }
              if (newStorageType !== comp.storageType) {
                console.log(`Updated storage type for ingredient "${comp.name}" in menu item "${menuItem.name}": ${comp.storageType} → ${newStorageType}`);
              }
              return {
                ...comp,
                unitCost: newUnitCost,
                totalCost: newUnitCost * comp.numberOfUnits,
                storageType: newStorageType,
              };
            }
          }
        } else if (comp.itemType === 'Component') {
          // Find the current component
          const currentComponent = components.find((c: any) => c.id === comp.sourceId);
          if (currentComponent) {
            const newUnitCost = currentComponent.unitCost;
            const newStorageType = currentComponent.storageType;
            if (newUnitCost !== comp.unitCost || newStorageType !== comp.storageType) {
              hasChanges = true;
              if (newUnitCost !== comp.unitCost) {
                totalCostChanges++;
                console.log(`Updated cost for component "${comp.name}" in menu item "${menuItem.name}": ${comp.unitCost} → ${newUnitCost}`);
              }
              if (newStorageType !== comp.storageType) {
                console.log(`Updated storage type for component "${comp.name}" in menu item "${menuItem.name}": ${comp.storageType} → ${newStorageType}`);
              }
              return {
                ...comp,
                unitCost: newUnitCost,
                totalCost: newUnitCost * comp.numberOfUnits,
                storageType: newStorageType,
              };
            }
          }
        }
        return comp;
      }) || [];
      
      if (hasChanges) {
        // Recalculate total cost and margin
        const totalCost = updatedComponents.reduce((sum: number, comp: any) => sum + comp.totalCost, 0) / menuItem.numberOfPiecesRecipe;
        const margin = menuItem.basePrice > 0 ? ((menuItem.basePrice - totalCost) / menuItem.basePrice) * 100 : 0;
        
        const updatedMenuItem = {
          ...menuItem,
          components: updatedComponents,
          totalCost,
          margin,
        };
        
        await kv.set(`menu-item:${menuItem.id}`, updatedMenuItem);
        updatedCount++;
      }
    }
    
    return c.json({ 
      success: true, 
      message: `Refreshed ${updatedCount} menu items with ${totalCostChanges} cost updates`,
      updatedCount,
      totalCostChanges
    });
  } catch (error) {
    console.error("Error refreshing menu item costs:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

  } catch (error) {

})
